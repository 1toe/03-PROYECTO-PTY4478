create table public."PRODUCTO" (
  id bigint generated by default as identity not null,
  nombre_producto character varying not null,
  val_energia real null,
  val_prote real null,
  val_grasas_tot real null,
  val_grasas_sat real null,
  val_monoinsatura real null,
  val_poliinsatura real null,
  val_grasas_trans real null,
  val_colesterol real null,
  val_carbono_dis real null,
  val_azucar_tot real null,
  val_fibra real null,
  val_sodio real null,
  descripcion character varying null,
  foto_url character varying null,
  constraint producto_pkey primary key (id)
) TABLESPACE pg_default;
